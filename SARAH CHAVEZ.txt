int leds[] = {2, 4, 5, 12, 13, 14, 15, 16, 17, 18}; 
int numLeds = 10;

void setup() {
  for (int i = 0; i < numLeds; i++) {
    pinMode(leds[i], OUTPUT);
    digitalWrite(leds[i], LOW); // inicializamos todos en modod apagado
  }
}

void loop() {
  //intÃ©rvalo de 2 seg
  for (int i = 0; i < numLeds; i++) {
    digitalWrite(leds[i], HIGH);   
    delay(2000);                   
    digitalWrite(leds[i], LOW);    
  }
}
